def post_claim_mark(params)
  ClaimMark.create! do |s|
    s.system_id = params[:system_id]
    s.claim_id = self.id
    s.posted_on = params[:posted_on]
    s.mark = params[:mark]
    s.approved = params[:approved]
  end
end
