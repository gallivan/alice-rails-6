def self.build_report(params, root)
  PackerReport.create! do |r|
    r.root = root unless root.blank?
    r.kind = 'FILL'
    r.fate = 'TODO'
    r.data = params
    r.posted_on = Date.today
  end
end

def self.fix2hash(payload)
  list = payload.chop.split("\x01")
  hash = {}
  list.each do |element|
    key, val = element.split('=')
    hash[key] = val
  end
  return hash
end

def self.pack
  conn_aacc = Bunny.new
  conn_aacc.start
  chnl_aacc = conn_aacc.create_channel

  q_aacc = chnl_aacc.queue("itd.abn.picked.all", :durable => true, :auto_delete => false)
  x_aacc = chnl_aacc.fanout("itd.abn.picked.all", :durable => true)
  q_aacc.bind(x_aacc)

  conn_norm = Bunny.new
  conn_norm.start
  chnl_norm = conn_norm.create_channel

  q_norm = chnl_norm.queue("itd.abn.packed.all", :durable => true, :auto_delete => false)
  x_norm = chnl_norm.fanout("itd.abn.packed.all", :durable => true)
  q_norm.bind(x_norm)

  q_aacc.subscribe(:manual_ack => true, :block => true) do |delivery_info, metadata, body|
    begin
      hash = self.fix2hash(body)

      report = build_report(hash, 'ITD_ABN')
      hash[:packer_report_id] = report.id

      Rails.logger.info("Packer: TODO handling PackerReport #{report.id}")
      puts "PackerReport: #{report.id}"

      if hash['35'] == '8'
        norm = Workers::Normalizer.normalize(hash)
        x_norm.publish(norm.to_json, :routing_key => q_norm.name)
      end

      chnl_aacc.ack(delivery_info.delivery_tag)
      report.update_attribute(:fate, 'DONE')
    rescue Exception => e
      chnl_aacc.reject(delivery_info.delivery_tag)
      report.update_attributes(fate: 'FAIL', goof_error: e.message, goof_trace: e.backtrace)
      msg = "Packer.pack handling of message failed with #{e}"
      Rails.logger.warn msg
      e.backtrace_locations.each do |location|
        Rails.logger.warn location
      end
    end
  end
end
